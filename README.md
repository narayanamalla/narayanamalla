```yaml
name: update-readme
on:
  schedule:
    - cron: '0 3 * * *' # runs daily at 03:00 UTC
  workflow_dispatch:

  update-readme:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

        run: |
          DATE=$(date '+%Y-%m-%d')
          sed -i "1s/.*/<!-- last updated: $DATE -->/" README.md || echo "<!-- last updated: $DATE -->" > README.md
      - name: Commit & Push
